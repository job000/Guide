Bruke Komponenter

Lær hvordan du importerer, bruker og sender data til komponenter i React/Next.js.

Hva er en komponent?

En komponent er en gjenbrukbar "byggestein" i React. Den er bare en funksjon som returnerer HTML (JSX).

// src/components/ui/button.tsx
export default function Button() {
  return (
    <button className="bg-blue-500 text-white px-4 py-2 rounded">
      Klikk meg
    </button>
  )
}

Slik bruker du en komponent

1

Importer komponenten

import Button from "@/components/ui/button"

@/ betyr src/ mappen. Dette er konfigurert i tsconfig.json.

2

Bruk komponenten som et HTML-element

<Button />

Komponenter starter alltid med stor bokstav.

Komplett eksempel

// src/app/page.tsx
import Button from "@/components/ui/button"

export default function HomePage() {
  return (
    <div>
      <h1>Velkommen</h1>
      <Button />
    </div>
  )
}

Sende data til komponenter (Props)

Props er hvordan du sender data fra forelder til barn-komponent. Tenk på det som funksjonsargumenter.

1. Komponent som tar imot props

// src/components/ui/button.tsx
interface ButtonProps {
  text: string          // Påkrevd
  color?: string        // Valgfri (?)
  onClick?: () => void  // Valgfri funksjon
}

export default function Button({ text, color = "blue", onClick }: ButtonProps) {
  return (
    <button
      className={`bg-${color}-500 text-white px-4 py-2 rounded`}
      onClick={onClick}
    >
      {text}
    </button>
  )
}

2. Bruke komponenten med props

// src/app/page.tsx
import Button from "@/components/ui/button"

export default function HomePage() {
  const handleClick = () => {
    alert("Du klikket!")
  }

  return (
    <div>
      {/* Enkel bruk */}
      <Button text="Klikk meg" />

      {/* Med farge */}
      <Button text="Grønn knapp" color="green" />

      {/* Med onClick */}
      <Button text="Alert" onClick={handleClick} />

      {/* Inline onClick */}
      <Button text="Console" onClick={() => console.log("Hei!")} />
    </div>
  )
}

Children - Innhold mellom tags

children er spesiell prop som lar deg legge innhold mellom komponent-tagsene.

1. Komponent med children

// src/components/ui/card.tsx
interface CardProps {
  title: string
  children: React.ReactNode  // Kan være hva som helst
}

export default function Card({ title, children }: CardProps) {
  return (
    <div className="bg-white rounded-lg shadow p-6">
      <h2 className="text-xl font-bold mb-4">{title}</h2>
      <div>{children}</div>
    </div>
  )
}

2. Bruke komponenten med innhold

import Card from "@/components/ui/card"

export default function Page() {
  return (
    <Card title="Min tittel">
      <p>Dette er innholdet i kortet.</p>
      <p>Du kan ha hva som helst her!</p>
      <button>En knapp</button>
    </Card>
  )
}

Vanlige mønstre

Komponent med conditional rendering

// Vis bare hvis betingelse er true
interface AlertProps {
  message: string
  show: boolean
}

export default function Alert({ message, show }: AlertProps) {
  if (!show) return null  // Returnerer ingenting

  return (
    <div className="bg-red-100 text-red-800 p-4 rounded">
      {message}
    </div>
  )
}

Komponent med liste-data

// Liste-komponent
interface User {
  id: number
  name: string
}

interface UserListProps {
  users: User[]
}

export default function UserList({ users }: UserListProps) {
  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name}</li>
      ))}
    </ul>
  )
}

// Bruk
const users = [
  { id: 1, name: "Ola" },
  { id: 2, name: "Kari" }
]

<UserList users={users} />

Komponent som sender data tilbake (callback)

// Søkefelt som sender søketekst tilbake
interface SearchProps {
  onSearch: (query: string) => void
}

export default function Search({ onSearch }: SearchProps) {
  return (
    <input
      placeholder="Søk..."
      onChange={(e) => onSearch(e.target.value)}
    />
  )
}

// Bruk i forelder
"use client"
import { useState } from "react"
import Search from "@/components/search"

export default function Page() {
  const [searchTerm, setSearchTerm] = useState("")

  return (
    <div>
      <Search onSearch={setSearchTerm} />
      <p>Du søker etter: {searchTerm}</p>
    </div>
  )
}

Vanlige feil

Feil 1: Glemmer å importere

// FEIL - Mangler import
export default function Page() {
  return <Button text="Hei" />  // Error: Button is not defined
}

// RIKTIG
import Button from "@/components/ui/button"

export default function Page() {
  return <Button text="Hei" />
}

Feil 2: Liten bokstav på komponent

// FEIL - React tror det er HTML-element
<button text="Hei" />

// RIKTIG - Stor bokstav
<Button text="Hei" />

Feil 3: Mangler key i liste

// FEIL - Mangler key
{users.map(user => <li>{user.name}</li>)}

// RIKTIG - Med key
{users.map(user => <li key={user.id}>{user.name}</li>)}

Feil 4: Feil path i import

// FEIL - Feil path eller filnavn
import Button from "@/components/Button"

// RIKTIG - Sjekk eksakt filnavn og path
import Button from "@/components/ui/button"
