Next.js Metadata API

Next.js 15 har innebygd metadata-støtte. Definer i layout.tsx eller page.tsx:

Global metadata (layout.tsx)

// src/app/layout.tsx
import type { Metadata } from "next"

export const metadata: Metadata = {
  // Basis
  title: {
    default: "Firmanavn - Kort beskrivelse",
    template: "%s | Firmanavn", // Brukes av undersider
  },
  description: "Kort beskrivelse av bedriften (150-160 tegn for Google)",

  // Søkemotor-innstillinger
  robots: {
    index: true,
    follow: true,
    googleBot: {
      index: true,
      follow: true,
      "max-video-preview": -1,
      "max-image-preview": "large",
      "max-snippet": -1,
    },
  },

  // Open Graph (Facebook, LinkedIn)
  openGraph: {
    type: "website",
    locale: "nb_NO",
    url: "https://firmanavn.no",
    siteName: "Firmanavn",
    title: "Firmanavn - Kort beskrivelse",
    description: "Beskrivelse for sosiale medier",
    images: [
      {
        url: "https://firmanavn.no/og-image.jpg",
        width: 1200,
        height: 630,
        alt: "Firmanavn",
      },
    ],
  },

  // Twitter
  twitter: {
    card: "summary_large_image",
    title: "Firmanavn",
    description: "Beskrivelse for Twitter",
    images: ["https://firmanavn.no/twitter-image.jpg"],
  },

  // Ikoner
  icons: {
    icon: "/favicon.ico",
    shortcut: "/favicon-16x16.png",
    apple: "/apple-touch-icon.png",
  },

  // Verifisering
  verification: {
    google: "din-google-verification-kode",
  },
}

Dynamisk metadata

// src/app/blogg/[slug]/page.tsx
import type { Metadata } from "next"

interface Props {
  params: { slug: string }
}

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  // Hent data fra database
  const post = await getPost(params.slug)

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      images: [post.image],
      type: "article",
      publishedTime: post.publishedAt,
      authors: [post.author],
    },
  }
}

Sitemap

Next.js 15 kan generere sitemap automatisk:

// src/app/sitemap.ts
import { MetadataRoute } from "next"

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = "https://firmanavn.no"

  // Statiske sider
  const staticPages = [
    "",
    "/om-oss",
    "/tjenester",
    "/kontakt",
    "/personvern",
  ].map((route) => ({
    url: `${baseUrl}${route}`,
    lastModified: new Date(),
    changeFrequency: "monthly" as const,
    priority: route === "" ? 1 : 0.8,
  }))

  // Dynamiske sider (f.eks. blogginnlegg)
  // const posts = await getPosts()
  // const postPages = posts.map((post) => ({
  //   url: `${baseUrl}/blogg/${post.slug}`,
  //   lastModified: new Date(post.updatedAt),
  //   changeFrequency: "weekly" as const,
  //   priority: 0.6,
  // }))

  return [...staticPages]
}

Sitemap blir tilgjengelig på: https://firmanavn.no/sitemap.xml

robots.txt

// src/app/robots.ts
import { MetadataRoute } from "next"

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: "*",
      allow: "/",
      disallow: ["/admin/", "/api/"],
    },
    sitemap: "https://firmanavn.no/sitemap.xml",
  }
}

Google Search Console

Verifiser domenet og send inn sitemap

Google Search Console

  1. Gå til search.google.com/search-console
  2. Legg til property (domenet ditt)
  3. Verifiser med DNS TXT-record eller HTML-tag
  4. Send inn sitemap: https://firmanavn.no/sitemap.xml
  5. Overvåk indeksering og feil

Ytelse påvirker SEO

Google bruker Core Web Vitals som ranking-faktor:

LCP

Largest Contentful Paint
< 2.5s

FID

First Input Delay
< 100ms

CLS

Cumulative Layout Shift
< 0.1

Test ytelse med:

  • PageSpeed Insights
  • web.dev/measure
  • Chrome DevTools → Lighthouse
