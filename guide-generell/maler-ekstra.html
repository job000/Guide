  // Bygg query basert p√• filtre
  let query = supabase
    .from("properties")
    .select(`
      *,
      real_estate_agents(name, phone, email, avatar_url)
    `)
    .eq("status", "for_sale")

  // Filtrer basert p√• s√∏keparametere
  if (searchParams.type) {
    query = query.eq("property_type", searchParams.type)
  }
  if (searchParams.minPrice) {
    query = query.gte("asking_price", parseInt(searchParams.minPrice) * 100)
  }
  if (searchParams.maxPrice) {
    query = query.lte("asking_price", parseInt(searchParams.maxPrice) * 100)
  }
  if (searchParams.minRooms) {
    query = query.gte("bedrooms", parseInt(searchParams.minRooms))
  }
  if (searchParams.city) {
    query = query.ilike("address_city", `%${searchParams.city}%`)
  }

  // Sortering
  switch (searchParams.sort) {
    case "price-asc":
      query = query.order("asking_price", { ascending: true })
      break
    case "price-desc":
      query = query.order("asking_price", { ascending: false })
      break
    case "newest":
      query = query.order("created_at", { ascending: false })
      break
    default:
      query = query.order("is_featured", { ascending: false })
        .order("created_at", { ascending: false })
  }

  const { data: properties } = await query

"use client"

import { useEffect, useRef } from "react"

interface PropertyMapProps {
  latitude: number
  longitude: number
  address: string
}

export default function PropertyMap({ latitude, longitude, address }: PropertyMapProps) {
  const mapRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    // Dette krever Google Maps API
    // Installer med: npm install @googlemaps/js-api-loader
    const loadMap = async () => {
      const { Loader } = await import("@googlemaps/js-api-loader")

      const loader = new Loader({
        apiKey: process.env.NEXT_PUBLIC_GOOGLE_MAPS_KEY!,
        version: "weekly",
      })

      const google = await loader.load()

      if (mapRef.current) {
        const map = new google.maps.Map(mapRef.current, {
          center: { lat: latitude, lng: longitude },
          zoom: 15,
        })

        new google.maps.Marker({
          position: { lat: latitude, lng: longitude },
          map,
          title: address,
        })
      }
    }

    loadMap()
  }, [latitude, longitude, address])

  return <div ref={mapRef} className="w-full h-96 rounded-lg" />
}
