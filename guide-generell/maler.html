// src/app/timeplan/page.tsx
import supabase from "@/config/supabase-config"

const DAYS = ["SÃ¸ndag", "Mandag", "Tirsdag", "Onsdag", "Torsdag", "Fredag", "LÃ¸rdag"]

interface ScheduleItem {
  id: string
  day_of_week: number
  start_time: string
  room: string
  fitness_class: {
    name: string
    duration_minutes: number
  }
  trainer: {
    name: string
  }
}

export default async function TimeplanPage() {
  const { data: schedule } = await supabase
    .from("class_schedule")
    .select(`
      *,
      fitness_class:fitness_classes(name, duration_minutes),
      trainer:trainers(name)
    `)
    .order("start_time")

  // Grupper etter ukedag
  const byDay = DAYS.map((_, index) => ({
    day: DAYS[index],
    classes: schedule?.filter((s: ScheduleItem) => s.day_of_week === index) || [],
  }))

  return (
    <div className="max-w-6xl mx-auto py-12 px-4">
      <h1 className="text-4xl font-bold text-center mb-12">Timeplan</h1>

      <div className="grid md:grid-cols-7 gap-4">
        {byDay.slice(1).concat(byDay.slice(0, 1)).map(({ day, classes }) => (
          <div key={day} className="bg-white rounded-lg p-4 shadow-sm">
            <h3 className="font-bold text-center mb-4 pb-2 border-b">{day}</h3>
            <div className="space-y-3">
              {classes.length === 0 ? (
                <p className="text-gray-400 text-sm text-center">Ingen timer</p>
              ) : (
                classes.map((item: ScheduleItem) => (
                  <div
                    key={item.id}
                    className="p-2 bg-blue-50 rounded text-sm"

                    <div className="font-medium">{item.fitness_class.name}</div>

                  </div>
                ))

            </div>
          </div>

    </div>
  )
}

CREATE OR REPLACE FUNCTION generate_order_number()
RETURNS TRIGGER AS $$
BEGIN
  NEW.order_number := 'ORD-' || TO_CHAR(NOW(), 'YYYYMMDD') || '-' ||
    LPAD(FLOOR(RANDOM() * 10000)::TEXT, 4, '0');
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_order_number
BEFORE INSERT ON orders
FOR EACH ROW EXECUTE FUNCTION generate_order_number();

// src/components/functional/product-card.tsx
import Image from "next/image"
import Link from "next/link"
import { IProduct } from "@/interfaces/product"
import AddToCartButton from "./add-to-cart-button"

interface ProductCardProps {
  product: IProduct
}

export default function ProductCard({ product }: ProductCardProps) {
  const isOnSale = product.compare_at_price && product.compare_at_price > product.price
  const discount = isOnSale
    ? Math.round((1 - product.price / product.compare_at_price!) * 100)
    : 0

  return (
    <div className="group bg-white rounded-lg overflow-hidden shadow-sm hover:shadow-md transition-shadow">
      <Link href={`/produkter/${product.slug}`}>
        <div className="relative aspect-square">
          <Image
            src={product.images[0] || "/placeholder.jpg"}
            alt={product.name}
            fill
            className="object-cover group-hover:scale-105 transition-transform"
            sizes="(max-width: 640px) 100vw, (max-width: 1024px) 50vw, 25vw"
          />
          {isOnSale && (
            <div className="absolute top-2 left-2 bg-red-500 text-white text-sm font-bold px-2 py-1 rounded">
              -{discount}%
            </div>
          )}
          {product.stock_quantity === 0 && (
            <div className="absolute inset-0 bg-black/50 flex items-center justify-center">
              <span className="text-white font-bold">Utsolgt</span>
            </div>
          )}
        </div>
      </Link>

      <div className="p-4">
        <Link href={`/produkter/${product.slug}`}>
          <h3 className="font-medium hover:text-blue-600 transition-colors">
            {product.name}
          </h3>
        </Link>

        <div className="flex items-center gap-2 mt-2">
          <span className="text-lg font-bold">
            {(product.price / 100).toFixed(0)} kr
          </span>
          {isOnSale && (
            <span className="text-sm text-gray-500 line-through">
              {(product.compare_at_price! / 100).toFixed(0)} kr
            </span>
          )}
        </div>

        <AddToCartButton product={product} className="mt-3 w-full" />
      </div>
    </div>
  )
}

"use client"

import { useRouter, useSearchParams } from "next/navigation"
import { IProductCategory } from "@/interfaces/product"

interface ProductFilterProps {
  categories: IProductCategory[]
}

export default function ProductFilter({ categories }: ProductFilterProps) {
  const router = useRouter()
  const searchParams = useSearchParams()

  const currentCategory = searchParams.get("kategori")
  const currentSort = searchParams.get("sorter") || "newest"
  const currentMinPrice = searchParams.get("minPris")
  const currentMaxPrice = searchParams.get("maxPris")

  const updateFilter = (key: string, value: string | null) => {
    const params = new URLSearchParams(searchParams.toString())
    if (value) {
      params.set(key, value)
    } else {
      params.delete(key)
    }
    router.push(`/produkter?${params.toString()}`)
  }

  return (
    <div className="bg-white p-4 rounded-lg shadow-sm space-y-6">
      {/* Kategorier */}
      <div>
        <h3 className="font-semibold mb-3">Kategorier</h3>
        <div className="space-y-2">
          <button
            onClick={() => updateFilter("kategori", null)}
            className={`block w-full text-left px-2 py-1 rounded ${
              !currentCategory ? "bg-blue-50 text-blue-600" : "hover:bg-gray-50"
            }`}
          >
            Alle produkter
          </button>
          {categories.map((cat) => (
            <button
              key={cat.id}
              onClick={() => updateFilter("kategori", cat.slug)}
              className={`block w-full text-left px-2 py-1 rounded ${
                currentCategory === cat.slug
                  ? "bg-blue-50 text-blue-600"
                  : "hover:bg-gray-50"
              }`}
            >
              {cat.name}
            </button>
          ))}
        </div>
      </div>

      {/* Sortering */}
      <div>
        <h3 className="font-semibold mb-3">Sorter etter</h3>
        <select
          value={currentSort}
          onChange={(e) => updateFilter("sorter", e.target.value)}

      </div>

      {/* Prisfilter */}

        <h3 className="font-semibold mb-3">Pris</h3>
