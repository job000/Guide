// src/app/(admin)/admin/meldinger/page.tsx
"use client"

import { useState, useEffect } from "react"
import { Mail, MailOpen, Trash2, ArrowLeft } from "lucide-react"
import toast from "react-hot-toast"

interface Message {
  id: number
  name: string
  email: string
  phone?: string
  message: string
  is_read: boolean
  created_at: string
}

export default function MessagesPage() {
  const [messages, setMessages] = useState<Message[]>([])
  const [selected, setSelected] = useState<Message | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    loadMessages()
  }, [])

  const loadMessages = async () => {
    const res = await fetch("/api/messages")
    const data = await res.json()
    setMessages(data)
    setLoading(false)
  }

  const markAsRead = async (id: number) => {
    await fetch(`/api/messages/${id}/read`, { method: "POST" })
    loadMessages()
    toast.success("Markert som lest")
  }

  // Mobil: Vis enten liste eller detalj
  // Desktop: Vis begge side om side
  return (
    <div className="p-4 md:p-8">
      <h1 className="text-2xl font-bold mb-6">Meldinger</h1>

      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        {/* Liste - skjul pa mobil hvis melding er valgt */}
        <div className={`bg-white rounded-xl border ${selected ? "hidden lg:block" : ""}`}>
          {loading ? (
            <div className="p-8 text-center text-gray-500">Laster...</div>
          ) : messages.length === 0 ? (
            <div className="p-8 text-center text-gray-500">Ingen meldinger</div>
          ) : (
            messages.map((msg) => (
              <div
                key={msg.id}
                onClick={() => setSelected(msg)}
                className={`p-4 border-b cursor-pointer hover:bg-gray-50
                  ${selected?.id === msg.id ? "bg-blue-50" : ""}
                  ${!msg.is_read ? "font-medium" : ""}`}
              >
                <div className="flex items-center gap-3">
                  {msg.is_read ? (
                    <MailOpen className="text-gray-400 shrink-0" size={20} />
                  ) : (
                    <Mail className="text-blue-500 shrink-0" size={20} />
                  )}
                  <div className="min-w-0">
                    <p className="truncate">{msg.name}</p>
                    <p className="text-sm text-gray-500 truncate">{msg.message}</p>
                  </div>
                </div>
              </div>
            ))
          )}
        </div>

        {/* Detalj */}
        <div className={`bg-white rounded-xl border ${!selected ? "hidden lg:flex items-center justify-center" : ""}`}>
          {selected ? (
            <div className="p-6">
              {/* Tilbake-knapp (mobil) */}
              <button
                onClick={() => setSelected(null)}
                className="lg:hidden flex items-center gap-2 text-gray-500 mb-4"
              >
                <ArrowLeft size={20} /> Tilbake
              </button>

              <div className="flex justify-between items-start mb-4">
                <div>
                  <h2 className="text-xl font-semibold">{selected.name}</h2>
                  <p className="text-gray-500">{selected.email}</p>
                  {selected.phone && <p className="text-gray-500">{selected.phone}</p>}
                </div>
                {!selected.is_read && (
                  <button
                    onClick={() => markAsRead(selected.id)}
                    className="px-3 py-1 bg-blue-100 text-blue-700 rounded-lg text-sm"
                  >
                    Marker lest
                  </button>
                )}
              </div>

              <div className="bg-gray-50 p-4 rounded-lg">
                <p className="whitespace-pre-wrap">{selected.message}</p>
              </div>

              <p className="text-sm text-gray-400 mt-4">
                Mottatt: {new Date(selected.created_at).toLocaleString("nb-NO")}
              </p>
            </div>
          ) : (
            <p className="text-gray-400">Velg en melding</p>
          )}
        </div>
      </div>
    </div>
  )
}

// src/app/timeplan/page.tsx
import { supabase } from "@/config/supabase-config"

const daysOfWeek = ["Mandag", "Tirsdag", "Onsdag", "Torsdag", "Fredag", "Lordag", "Sondag"]

async function getSchedule() {
  const { data } = await supabase
    .from("schedule")
    .select(`
      *,
      class:classes(*),
      instructor:instructors(name)
    `)
    .eq("is_active", true)
    .order("start_time")

  return data || []
}

export default async function SchedulePage() {
  const schedule = await getSchedule()

  // Grupper etter dag
  const byDay = daysOfWeek.map((day, index) => ({
    name: day,
    classes: schedule.filter(s => s.day_of_week === index)
  }))

  return (
    <div className="py-12">
      <div className="max-w-6xl mx-auto px-4">
        <h1 className="text-3xl font-bold text-center mb-8">Timeplan</h1>

        {/* Mobil: Vertikal liste */}
        <div className="md:hidden space-y-6">
          {byDay.map((day) => (
            <div key={day.name}>
              <h2 className="font-bold text-lg mb-3 pb-2 border-b">{day.name}</h2>
              {day.classes.length === 0 ? (
                <p className="text-gray-500 text-sm">Ingen timer</p>
              ) : (
                <div className="space-y-2">
                  {day.classes.map((item) => (
                    <div key={item.id} className="flex items-center justify-between
                      bg-white p-3 rounded-lg border">
                      <div>
                        <p className="font-medium">{item.class?.name}</p>
                        <p className="text-sm text-gray-500">
                          {item.instructor?.name} ‚Ä¢ {item.class?.duration} min
                        </p>
                      </div>
                      <div className="text-right">
                        <p className="font-bold">{item.start_time.slice(0, 5)}</p>
                        <button className="text-xs text-blue-600">Book</button>
                      </div>
                    </div>
                  ))}
                </div>
              )}
            </div>
          ))}
        </div>

        {/* Desktop: Tabell */}
        <div className="hidden md:block overflow-x-auto">
          <table className="w-full">
            <thead>
              <tr>
                <th className="p-2"></th>
                {daysOfWeek.map(day => (
                  <th key={day} className="p-2 text-center">{day}</th>
                ))}
              </tr>
            </thead>
            <tbody>
              {/* Generer rader for hver time (06:00 - 21:00) */}
              {Array.from({ length: 16 }, (_, i) => i + 6).map(hour => (
                <tr key={hour} className="border-t">
                  <td className="p-2 text-sm text-gray-500">{hour}:00</td>
                  {daysOfWeek.map((_, dayIndex) => {
                    const classAtTime = schedule.find(
                      s => s.day_of_week === dayIndex &&
                      parseInt(s.start_time) === hour
                    )
                    return (
                      <td key={dayIndex} className="p-1">
                        {classAtTime && (
                          <div className="bg-blue-100 text-blue-800 p-2 rounded text-xs">
                            <p className="font-medium">{classAtTime.class?.name}</p>
                            <p>{classAtTime.instructor?.name}</p>
                          </div>
                        )}
                      </td>
                    )
                  })}
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  )
}

                >
                  <div className="flex justify-between items-center">
                    <div>
                      <p className="font-medium">{service.name}</p>
                      <p className="text-sm text-gray-500">{service.duration} min</p>
                    </div>
                    <p className="font-bold">{service.price} kr</p>
                  </div>
                </button>
              ))}
            </div>
          </div>
        )}

        {/* Steg 2: Velg dato og tid */}
        {step === "datetime" && (
          <div className="bg-white rounded-2xl p-6 shadow-sm">
            <button onClick={() => setStep("service")} className="flex items-center text-gray-500 mb-4">
              <ChevronLeft size={20} /> Tilbake
            </button>

            <h2 className="text-xl font-bold mb-4">Velg dato og tid</h2>

            {/* Enkel datovelger */}
            <div className="mb-6">
              <input
                type="date"
                onChange={(e) => setSelectedDate(new Date(e.target.value))}
                min={new Date().toISOString().split("T")[0]}
                className="w-full p-3 border rounded-xl"
              />
            </div>

            {/* Tidsvalg */}
            {selectedDate && (
              <div>
                <p className="text-sm text-gray-500 mb-2">Ledige tider</p>
                <div className="grid grid-cols-3 gap-2">
                  {availableTimes.map((time) => (
                    <button
                      key={time}
                      onClick={() => {
                        setSelectedTime(time)
                        setStep("confirm")
                      }}
                      className={`p-3 rounded-lg border-2 font-medium
                        ${selectedTime === time
                          ? "border-blue-600 bg-blue-50"
                          : "border-gray-200 hover:border-blue-300"}`}
                    >
                      {time}
                    </button>
                  ))}
                </div>
              </div>
            )}
          </div>
        )}

        {/* Steg 3: Bekreft */}
        {step === "confirm" && (
          <div className="bg-white rounded-2xl p-6 shadow-sm">
            <button onClick={() => setStep("datetime")} className="flex items-center text-gray-500 mb-4">
              <ChevronLeft size={20} /> Tilbake
            </button>

            <h2 className="text-xl font-bold mb-4">Bekreft bestilling</h2>

            {/* Oppsummering */}
            <div className="bg-gray-50 rounded-xl p-4 mb-6">
              <p><strong>{selectedService?.name}</strong></p>
              <p className="text-gray-500">
                {selectedDate?.toLocaleDateString("nb-NO")} kl. {selectedTime}
              </p>
              <p className="text-lg font-bold mt-2">{selectedService?.price} kr</p>
            </div>

            {/* Kontaktinfo */}
            <div className="space-y-4">
              <input
                type="text"
                placeholder="Ditt navn"
                value={formData.name}
                onChange={(e) => setFormData({ ...formData, name: e.target.value })}
                className="w-full p-3 border rounded-xl"
              />
              <input
                type="email"
                placeholder="E-post"
                value={formData.email}
                onChange={(e) => setFormData({ ...formData, email: e.target.value })}
                className="w-full p-3 border rounded-xl"
              />
              <input
                type="tel"
                placeholder="Telefon"
                value={formData.phone}
                onChange={(e) => setFormData({ ...formData, phone: e.target.value })}
                className="w-full p-3 border rounded-xl"
              />
            </div>

            <button
              onClick={handleSubmit}
              className="w-full mt-6 bg-blue-600 text-white py-4 rounded-xl
                font-bold hover:bg-blue-700 flex items-center justify-center gap-2"
            >
              <Check size={20} /> Bekreft bestilling
            </button>
          </div>
        )}
      </div>
    </div>
  )
}

FAQ-side med sok

// src/app/hjelp/page.tsx
"use client"

import { useState, useEffect } from "react"
import { Search, ChevronDown, ChevronUp } from "lucide-react"
import Link from "next/link"

interface FAQ {
  id: number
  question: string
  answer: string
  category: { name: string }
}

export default function HelpPage() {
  const [faqs, setFaqs] = useState<FAQ[]>([])
  const [search, setSearch] = useState("")
  const [openId, setOpenId] = useState<number | null>(null)

  useEffect(() => {
    fetch("/api/faqs").then(r => r.json()).then(setFaqs)
  }, [])

  const filtered = faqs.filter(faq =>
    faq.question.toLowerCase().includes(search.toLowerCase()) ||
    faq.answer.toLowerCase().includes(search.toLowerCase())
  )

  return (
    <div className="py-12">
      <div className="max-w-3xl mx-auto px-4">
        {/* Header */}
        <div className="text-center mb-8">
          <h1 className="text-3xl font-bold mb-4">Hvordan kan vi hjelpe?</h1>

          {/* Sokefelt */}
          <div className="relative max-w-xl mx-auto">
            <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-gray-400" size={20} />
            <input
              type="text"
              placeholder="Sok i hjelpesenter..."
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              className="w-full pl-12 pr-4 py-4 rounded-xl border-2 border-gray-200
                focus:border-blue-500 focus:outline-none text-lg"
            />
          </div>
        </div>

        {/* Kategorier (hurtiglenker) */}
        <div className="grid grid-cols-2 md:grid-cols-4 gap-4 mb-8">
          {["Kom i gang", "Betaling", "Konto", "Teknisk"].map((cat) => (
            <button
              key={cat}
              onClick={() => setSearch(cat)}
              className="p-4 bg-white rounded-xl border hover:border-blue-500 transition-colors"
            >
              <p className="font-medium">{cat}</p>
            </button>
          ))}
        </div>

        {/* FAQ-liste */}
        <div className="space-y-3">
          {filtered.map((faq) => (
            <div key={faq.id} className="bg-white rounded-xl border overflow-hidden">
              <button
                onClick={() => setOpenId(openId === faq.id ? null : faq.id)}
                className="w-full p-4 flex items-center justify-between text-left"
              >
                <span className="font-medium pr-4">{faq.question}</span>
                {openId === faq.id ? (
                  <ChevronUp className="shrink-0 text-gray-400" size={20} />
                ) : (
                  <ChevronDown className="shrink-0 text-gray-400" size={20} />
                )}
              </button>
              {openId === faq.id && (
                <div className="px-4 pb-4 text-gray-600">
                  {faq.answer}
                </div>
              )}
            </div>
          ))}
        </div>

        {/* Fant du ikke svar? */}
        <div className="mt-8 text-center bg-blue-50 rounded-xl p-6">
          <p className="text-gray-600 mb-4">Fant du ikke det du lette etter?</p>
          <Link
            href="/kontakt"
            className="inline-block bg-blue-600 text-white px-6 py-3 rounded-lg font-medium hover:bg-blue-700"
          >
            Kontakt oss
          </Link>
        </div>
      </div>
    </div>
  )
}
