Button-komponent

En fleksibel knapp med varianter (default, outline, ghost, destructive) og størrelser. Bruker class-variance-authority for å håndtere varianter. 

// src/components/ui/button.tsx
import * as React from "react"
import { cva, type VariantProps } from "class-variance-authority"
import { cn } from "@/lib/utils"

const buttonVariants = cva(
  "inline-flex items-center justify-center gap-2 rounded-md text-sm font-medium transition-colors focus-visible:outline-none focus-visible:ring-2 disabled:pointer-events-none disabled:opacity-50",
  {
    variants: {
      variant: {
        default: "bg-primary text-white hover:bg-primary/90",
        outline: "border border-gray-300 bg-white hover:bg-gray-50",
        ghost: "hover:bg-gray-100",
        destructive: "bg-red-500 text-white hover:bg-red-600",
      },
      size: {
        default: "h-10 px-4 py-2",
        sm: "h-8 px-3 text-xs",
        lg: "h-12 px-6",
        icon: "h-10 w-10",
      },
    },
    defaultVariants: {
      variant: "default",
      size: "default",
    },
  }
)

interface ButtonProps
  extends React.ButtonHTMLAttributes<HTMLButtonElement>,
    VariantProps<typeof buttonVariants> {}

const Button = React.forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant, size, ...props }, ref) => {
    return (
      <button
        className={cn(buttonVariants({ variant, size, className }))}
        ref={ref}
        {...props}
      />
    )
  }
)
Button.displayName = "Button"

export { Button, buttonVariants }

Bruk:

<Button>Standard</Button>
<Button variant="outline">Outline</Button>
<Button variant="destructive">Slett</Button>
<Button size="lg">Stor knapp</Button>
<Button disabled>Deaktivert</Button>

Input-komponent

Et stilrent tekstfelt med fokus-ring og placeholder-styling. Brukes i alle skjemaer. 

// src/components/ui/input.tsx
import * as React from "react"
import { cn } from "@/lib/utils"

const Input = React.forwardRef<
  HTMLInputElement,
  React.InputHTMLAttributes<HTMLInputElement>
>(({ className, type, ...props }, ref) => {
  return (
    <input
      type={type}
      className={cn(
        "flex h-10 w-full rounded-md border border-gray-300 bg-white px-3 py-2 text-sm placeholder:text-gray-400 focus:outline-none focus:ring-2 focus:ring-primary focus:border-transparent disabled:cursor-not-allowed disabled:opacity-50",
        className
      )}
      ref={ref}
      {...props}
    />
  )
})
Input.displayName = "Input"

export { Input }

Kort-komponent (Card)

En generisk boks for å vise innhold med tittel, beskrivelse og valgfritt barn-innhold. Perfekt for produkter, artikler, etc. 

// src/components/functional/card.tsx
interface CardProps {
  title: string
  description?: string
  children?: React.ReactNode
  className?: string
}

export default function Card({ title, description, children, className = "" }: CardProps) {
  return (
    <div className={`bg-white rounded-lg border border-gray-200 p-5 shadow-sm ${className}`}>
      <h3 className="text-lg font-semibold">{title}</h3>
      {description && <p className="text-gray-600 mt-2">{description}</p>}
      {children && <div className="mt-4">{children}</div>}
    </div>
  )
}

Dashboard-kort

Et statistikk-kort for dashboards. Viser tall med ikon og fargetema. Perfekt for "Totale salg", "Nye brukere", etc. 

// src/components/functional/dashboard-card.tsx
import React from "react"

interface Props {
  title: string
  value: number
  icon: React.ReactNode
  className?: string
  valueClassName?: string
}

export default function DashboardCard({
  title,
  value,
  icon,
  className = "",
  valueClassName = "",
}: Props) {
  return (
    <div className={`p-5 rounded-lg border ${className}`}>
      <div className="flex items-center justify-between">
        <div>
          <p className="text-gray-600 text-sm font-medium">{title}</p>
          <p className={`text-3xl font-bold mt-1 ${valueClassName}`}>{value}</p>
        </div>
        <div className="p-3 rounded-full bg-white shadow-sm">
          {icon}
        </div>
      </div>
    </div>
  )
}

Utloggingsknapp

En ferdig utloggingsknapp som fjerner token-cookies og omdirigerer til login-siden. Bruker js-cookie og react-hot-toast. 

// src/components/functional/logout-btn.tsx
"use client"

import { useRouter } from "next/navigation"
import Cookies from "js-cookie"
import toast from "react-hot-toast"
import { Button } from "@/components/ui/button"
import { LogOut } from "lucide-react"

export default function LogoutBtn() {
  const router = useRouter()

  const handleLogout = () => {
    Cookies.remove("token")
    Cookies.remove("role")
    toast.success("Du er nå logget ut")
    router.push("/login")
  }

  return (
    <Button variant="ghost" onClick={handleLogout}>
      <LogOut className="h-4 w-4 mr-2" />
      Logg ut
    </Button>
  )
}

cn() hjelpefunksjon

En utility-funksjon som kombinerer CSS-klasser. Håndterer betinget styling og slår sammen Tailwind-klasser korrekt. Må-ha for alle prosjekter! 

// src/lib/utils.ts
import { clsx, type ClassValue } from "clsx"
import { twMerge } from "tailwind-merge"

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

// Bruk:
<div className={cn(
  "px-4 py-2 rounded",
  isActive && "bg-blue-500 text-white",
  !isActive && "bg-gray-100",
  className
)}>
  Innhold
</div>
